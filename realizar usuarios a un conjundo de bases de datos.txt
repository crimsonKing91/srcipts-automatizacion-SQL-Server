-- 2. ASIGNACIÓN DE PERMISOS MASIVOS (Nivel Bases de Datos)
DECLARE @Usuario NVARCHAR(128) = 'drenc';
DECLARE @DBName NVARCHAR(128);
DECLARE @SQL NVARCHAR(MAX);

-- Filtramos específicamente por las bases que contienen OLSA
DECLARE db_cursor CURSOR FOR 
SELECT name 
FROM sys.databases 
WHERE name NOT IN ('master', 'model', 'msdb', 'tempdb') 
  AND state = 0 
  AND name LIKE '%drenc%' -- <--- Filtro específico para este grupo

OPEN db_cursor;
FETCH NEXT FROM db_cursor INTO @DBName;

WHILE @@FETCH_STATUS = 0
BEGIN
    SET @SQL = 'USE [' + @DBName + '];
                IF NOT EXISTS (SELECT * FROM sys.database_principals WHERE name = ''' + @Usuario + ''')
                BEGIN
                    CREATE USER [' + @Usuario + '] FOR LOGIN [' + @Usuario + '] WITH DEFAULT_SCHEMA = [dbo];
                END
                ELSE
                BEGIN
                    ALTER USER [' + @Usuario + '] WITH DEFAULT_SCHEMA = [dbo];
                END
                ALTER ROLE db_owner ADD MEMBER [' + @Usuario + '];';
    
    PRINT 'Permisos de propietario asignados en: ' + @DBName;
    EXEC sp_executesql @SQL;

    FETCH NEXT FROM db_cursor INTO @DBName;
END

CLOSE db_cursor;
DEALLOCATE db_cursor;
PRINT 'Proceso completado para el usuario.';
